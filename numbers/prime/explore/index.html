Prime Check

Enter a natural number to check whether it is prime.

View Code 

// JavaScript
function isPrime(n) {
  if (n <= 1) return false;
  if (n === 2) return true;
  if (n % 2 === 0) return false;

  for (let i = 3; i * i <= n; i += 2) {
    if (n % i === 0) return false;
  }
  return true;
}

List Primes in a Range

Enter a start and end value to list all prime numbers in that range.

View Code 

// JavaScript
function listPrimesInRange(a, b) {
  const primes = [];

  for (let n = a; n <= b; n++) {
    if (isPrime(n)) {
      primes.push(n);
    }
  }

  return primes;
}

Count Primes up to N

Enter a number N to count how many primes are less than or equal to N.

View Code 

// JavaScript
function countPrimesUpTo(n) {
  let count = 0;

  for (let i = 2; i <= n; i++) {
    if (isPrime(i)) {
      count++;
    }
  }

  return count;
}
